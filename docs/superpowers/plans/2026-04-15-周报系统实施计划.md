```python
from config import DB_CONFIG

def init_database():
    # 创建数据库连接
    conn = pymysql.connect(**DB_CONFIG)
    cursor = conn.cursor()

    # 创建tasks表
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS tasks (
            id INT PRIMARY KEY AUTO_INCREMENT,
            name VARCHAR(200) NOT NULL,
            category VARCHAR(50) NOT NULL,
            priority VARCHAR(10) DEFAULT '中',
            status VARCHAR(20) DEFAULT '未开始',
            plan_start DATETIME,
            plan_end DATETIME,
            progress_note TEXT,
            pause_reason TEXT,
            sort_order INT DEFAULT 0,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
            updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
        )
    """)

    conn.commit()
    cursor.close()
    conn.close()
    print("数据库初始化完成")

if __name__ == '__main__':
    init_database()
```

```python
@app.route('/api/tasks', methods=['GET'])
def get_tasks():
    conn = get_db_connection()
    cursor = conn.cursor(pymysql.cursors.DictCursor)
    cursor.execute("SELECT * FROM tasks ORDER BY category, sort_order, id")
    tasks = cursor.fetchall()
    cursor.close()
    conn.close()
    return jsonify(tasks)

@app.route('/api/tasks', methods=['POST'])
def create_task():
    data = request.json
    conn = get_db_connection()
    cursor = conn.cursor()

    sql = """INSERT INTO tasks (name, category, priority, status, plan_start, plan_end)
             VALUES (%s, %s, %s, '未开始', %s, %s)"""
    cursor.execute(sql, (
        data['name'],
        data['category'],
        data.get('priority', '中'),
        data.get('plan_start'),
        data.get('plan_end')
    ))

    conn.commit()
    task_id = cursor.lastrowid
    cursor.close()
    conn.close()
    return jsonify({'id': task_id, 'message': '任务创建成功'})

@app.route('/api/tasks/<int:task_id>', methods=['PUT'])
def update_task(task_id):
    data = request.json
    conn = get_db_connection()
    cursor = conn.cursor()

    sql = """UPDATE tasks SET
             name=%s, category=%s, priority=%s, status=%s,
             plan_start=%s, plan_end=%s, progress_note=%s, pause_reason=%s
             WHERE id=%s"""
    cursor.execute(sql, (
        data['name'],
        data['category'],
        data.get('priority', '中'),
        data.get('status', '未开始'),
        data.get('plan_start'),
        data.get('plan_end'),
        data.get('progress_note'),
        data.get('pause_reason'),
        task_id
    ))

    conn.commit()
    cursor.close()
    conn.close()
    return jsonify({'message': '任务更新成功'})

@app.route('/api/tasks/<int:task_id>', methods=['DELETE'])
```

```python
        sql = """SELECT * FROM tasks

        today = datetime.now().strftime('%Y-%m-%d')

                 WHERE DATE(plan_start) = %s
                 ORDER BY category, sort_order, id"""
```

```python
@app.route('/api/report/generate', methods=['POST'])
def generate_report():
    data = request.json
    week_start = data.get('week_start')  # 格式: YYYY-MM-DD
    week_end = data.get('week_end')

    # 计算周次
    week_start_date = datetime.strptime(week_start, '%Y-%m-%d')
    week_end_date = datetime.strptime(week_end, '%Y-%m-%d')
    week_label = week_start_date.strftime('%Y%m%d') + '-' + week_end_date.strftime('%Y%m%d')

    conn = get_db_connection()
    cursor = conn.cursor(pymysql.cursors.DictCursor)

    # 查询已完成任务
    sql = """SELECT * FROM tasks
             WHERE status = '已完成'
             AND plan_end >= %s
             AND plan_end <= %s
             ORDER BY category, sort_order, id"""
    cursor.execute(sql, (week_start, week_end + ' 23:59:59'))
    completed_tasks = cursor.fetchall()

    # 按分类分组
    categories = ['教学', '竞赛', '就业', '科研', '项目', '职能组', '院校支撑']
    grouped = {cat: [] for cat in categories}
    for task in completed_tasks:
        cat = task['category']
        if cat in grouped:
            grouped[cat].append(task)

    # 创建Excel
    wb = Workbook()
    ws = wb.active
    ws.title = '工作周报'

    # 标题行
    ws['A1'] = '工作周报'
    ws['A2'] = '汇报人'
    ws['C2'] = '汇报时间段'
    ws['A3'] = '所属项目'
    ws['C3'] = '提交时间'

    # 基本信息
    ws['B2'] = '王弯弯'
    ws['D2'] = f"{week_start} 至 {week_end}"
    ws['B3'] = '河工大'
    ws['D3'] = = datetime.now().strftime('%Y-%m-%d %H:%M')

    # 表头
    ws['A4'] = '工作分类'
    ws['B4'] = '本周计划'
    ws['C4'] = '本周完成情况'
    ws['D4'] = '下周工作计划'
    ws['E4'] = '待协调、支撑事项'

    # 分类数据行
    row = 5
    for cat in categories:
        ws[f'A{row}'] = cat
        tasks_in_cat = grouped.get(cat, [])
        if tasks_in_cat:
            # 填写已完成任务
            task_texts = []
            for t in tasks_in_cat:
                name = t['name']
                note = t.get('progress_note', '')
                if note:
                    task_texts.append(f"{name}：{note}")
                else:
                    task_texts.append(name)
            ws[f'C{row}'] = '\n'.join(task_texts)
        row += 1

    # 保存文件
    filename = f"{week_label}-周报.xlsx"
    filepath = f"/tmp/{filename}"
    wb.save(filepath)

    cursor.close()
    conn.close()

    return send_file(filepath, as_attachment=True, download_name=filename)
```

```html
function renderTasks(tasks) {
    const container = document.getElementById('taskList');
    const categories = ['教学', '竞赛', '就业', '科研', '项目', '职能组', '院校支撑'];

                        <th>计划时间</th>
```

```html
// 计算上周的周一和周日
function getLastWeekDates() {
    const today = new Date();
    const dayOfWeek = today.getDay() || 7;
    const monday = new Date(today);
    monday.setDate(today.getDate() - dayOfWeek - 6);
    const sunday = new Date(monday);
    sunday.setDate(monday.getDate() + 6);

    return {
        start: monday.toISOString().split('T')[0],
        end: sunday.toISOString().split('T')[0]
    };
}

function showWeekPreview() {
    const week = getLastWeekDates();
    document.getElementById('weekPreview').innerHTML =
        `上周周期：${week.start} 至 ${week.end}`;
}
showWeekPreview();

async function generateReport() {
    const week = getLastWeekDates();

    const response = await fetch('/api/report/generate', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({
            week_start: week.start,
            week_end: week.end
        })
    });

    // 下载文件
    const blob = await response.blob();
    const url = window.URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = response.headers.get('Content-Disposition')?.split('filename=')[1] || '周报.xlsx';
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    window.URL.revokeObjectURL(url);
}
```
